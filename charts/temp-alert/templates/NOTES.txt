{{- if eq .Values.service.type "LoadBalancer" }}

  1. Get the LoadBalancer IP:

     kubectl get service {{ include "alert-manager.fullname" . }} -n {{ include "alert-manager.namespace" . }}

  2. Access the Alert Manager UI:

     export SERVICE_IP=$(kubectl get svc {{ include "alert-manager.fullname" . }} -n {{ include "alert-manager.namespace" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
     echo "Alert Manager UI: http://$SERVICE_IP:{{ .Values.service.port }}"

{{- else if eq .Values.service.type "NodePort" }}

  1. Get the Node IP and Port:

     export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
     export NODE_PORT=$(kubectl get svc {{ include "alert-manager.fullname" . }} -n {{ include "alert-manager.namespace" . }} -o jsonpath='{.spec.ports[0].nodePort}')
     
  2. Access the Alert Manager UI:

     echo "Alert Manager UI: http://$NODE_IP:$NODE_PORT"

{{- else }}

  1. Port-forward to access locally:

     kubectl port-forward -n {{ include "alert-manager.namespace" . }} svc/{{ include "alert-manager.fullname" . }} 9093:{{ .Values.service.port }}

  2. Access the Alert Manager UI:

     echo "Alert Manager UI: http://localhost:9093"
