  export NODE_IP=$(kubectl get nodes --namespace {{ include "node-red.namespace" . }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Node-RED URL: http://$NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status by running:
        
        kubectl --namespace {{ include "node-red.namespace" . }} get services -w {{ include "node-red.fullname" . }}
  
  export SERVICE_IP=$(kubectl get svc --namespace {{ include "node-red.namespace" . }} {{ include "node-red.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Node-RED URL: http://$SERVICE_IP:{{ .Values.service.port }}"
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ include "node-red.namespace" . }} -l "app.kubernetes.io/name={{ include "node-red.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ include "node-red.namespace" . }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Node-RED URL: http://127.0.0.1:1880"
  kubectl --namespace {{ include "node-red.namespace" . }} port-forward $POD_NAME 1880:$CONTAINER_PORT
{{- end }}

3. Quick Status Check:

   # Check deployment status
   kubectl get all -n {{ include "node-red.namespace" . }}
   
   # View logs
   kubectl logs -f -n {{ include "node-red.namespace" . }} -l app=node-red
   
   # Check resource usage
   kubectl top pod -n {{ include "node-red.namespace" . }}

4. Common Tasks:

   # Backup your flows
   kubectl exec -n {{ include "node-red.namespace" . }} -it deployment/{{ include "node-red.fullname" . }} -- tar czf /data/backup.tar.gz /data/flows.json /data/.config.json
   kubectl cp {{ include "node-red.namespace" . }}/$(kubectl get pod -n {{ include "node-red.namespace" . }} -l app=node-red -o jsonpath='{.items[0].metadata.name}'):/data/backup.tar.gz ./node-red-backup.tar.gz
   
   # Restart Node-RED
   kubectl rollout restart deployment/{{ include "node-red.fullname" . }} -n {{ include "node-red.namespace" . }}
   
   # Scale to zero (maintenance)
   kubectl scale deployment/{{ include "node-red.fullname" . }} --replicas=0 -n {{ include "node-red.namespace" . }}
   
   # Scale back up
   kubectl scale deployment/{{ include "node-red.fullname" . }} --replicas=1 -n {{ include "node-red.namespace" . }}

   # Check events
   kubectl get events -n {{ include "node-red.namespace" . }} --sort-by=.lastTimestamp
